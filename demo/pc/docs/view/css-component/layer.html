弹窗

依赖文件

  • cfec.css
  • jquery.js
  • layer.js
  • artTemplate.js

使用方法

1.先使用requireJS导入所需的依赖js文件
2.拷贝对应代码进去，就可以实现。
3.js API参考多对应的js模块的文档
4.layer插件的icon对应 1：ok 2：error 3：警告

实例

在新窗口查看效果

单行文字代码

             
                require(['jquery','layer','artTemplate'],function($,layer,template){
                           $(".j-alert").click(function(){
                             var html_tpl= template('layer_base', new Object());
                             var layer1= layer.open({
                                  area: "420px",
                                  fix: false,
                                  shadeClose: true,
                                  title: "提示",
                                  btn: ['确认', '取消'],
                                  content:html_tpl
                                });
                          })
                })
                    <!--基础弹窗[-->
                        <script id="layer_base" type="text/html">
                            <p class="tac">我是提示信息.</p>
                        </script>
                    <!--基础弹窗]-->

带图标弹窗代码

             
                require(['jquery','layer','artTemplate'],function($,layer,template){
                          $(".j-alert-status-multi").click(function(){
                                var html_tpl= template('layer_status_multi', new Object());
                                var layer2= layer.open({
                                    area: "420px",
                                    fix: false,
                                    shadeClose: true,
                                    title: "提示",
                                    icon: 1,
                                    btn: ['确认', '取消'],

                            });
                });

                       </script>
