selectCategory ,类目选择器

selectCategory实现商品筛选功能。

SuperSlide组件文档

实例

依赖:

  • require.js
  • jquery.js

代码块

  require(['jquery','selectCategory'], function ($，selectCategory){
       var a = new selectCategory({
            id: $('#selectBox'),         //绑定的容器
            api: [api.COMMON.getCategorysByIndusId, api.COMMON.getCategorysByParentId],     //数据接口,第二个参数是获取下级类目的接口
            isOneLevel: false            //级别
        });
  }

相关功能

  • 获取类目的路径，绑定在<span id="selCategorys"></span>

    代码块

 a.on('cateChang', function(id) {
            var text = "";
            for (var i = 0; i < a.crumbs.length; i++) {
                if (i == (a.crumbs.length - 1)) {
                    lastCategoryId = a.crumbs[i].categoryId;
                }
                if (i == 0) {
                    text += a.crumbs[i].categoryName;
                    if (i != (a.crumbs.length - 1)) {
                        text += " >> ";
                    }
                    continue;
                } else {
                    text += a.crumbs[i].categoryName;
                    if (i != (a.crumbs.length - 1)) {
                        text += " >> ";
                    }
                }
            }
            document.getElementById("selCategorys").innerHTML = text;
        });
  • 获取选择过的类目.通过select标签存放选择过的路径，

    代码块

<select name="onceCategory" id="onceCategoryId">
</select>

$("#onceCategoryId").on('change', function() {
            lastCategoryId = $(this).val();
            if (lastCategoryId == null || lastCategoryId == "") {
                return;
            }
            var idsArr = lastCategoryId.split(',');
            a.initCrumbs(idsArr);
            lastCategoryId = idsArr[idsArr.length - 1];
            if (lastCategoryId != '') {
                document.getElementById("selCategorys").innerHTML = $(this).find("option:selected").text();
            } else {
                document.getElementById("selCategorys").innerHTML = "";
            }
        });
