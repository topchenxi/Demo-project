## js:

define(['jquery', 'jqzoom'], function($) $(".slideTxtBox").slide({ mainCell: ".bd ul", titCell: ".hd i", effect: "fade", autoPlay: true, interTime: "5000", delayTime: "1000", titOnClassName: "pointer-active" }); });



### 基础参数

$(".slideTxtBox").slide({ autoPage: false, //程序自动分页需结合titCell使用，若为true，则titCell为导航元素的包裹层对象。 mainCell: ".bd ul", //导航元素对象。 titCell: ".hd i", //切换元素的包裹层对象。 effect: "fade", //动画效果 fade(渐显) top(滚动) left(左滚动) topLoop(上循环滚动) 。 //leftLoop(左循环滚动) topMarquee(上无缝循环滚动) leftMarquee(左无缝循环滚动)。 //fold(淡入淡出)(ps:v2.0) slideDown(下拉效果)(ps:v2.1)。

autoPlay: true,           //自动运行
interTime: "5000",        //毫秒,自动运行间隔。
delayTime: "1000",        //毫秒；切换效果持续时间。
trigger:"mouseover",      //titCell触发方式: mouseover(鼠标移过触发) click(鼠标点击触发)。
scroll:"1"                //每次滚动个数。
vis:"1"                   //visible缩写，mainCell的可视范围个数。
titOnClassName: ".class"  //当前titCell位置自动增加的class名称。
easing:"swing"            //缓动效果,使效果更流畅。

});
